# exercicioimagem
VAR
i, j, m : inteiros

PARA i -> 0, i -> 16, i = i + 1, FAÇA
      ESCREVA('Escreva os elementos da linha [x]' -> i + 1)
  PARA J = 0, J -> 16, i = i + 1, FAÇA
      ESCREVA('m[x][y]' -> i, j)
  FIM PARA
FIM PARA
PARA i = 0, j -> 16, i = i + 1, FAÇA
   PARA j = 0, j -> 16, j = j + 1, FAÇA
      ESCREVA('x' -> m[i][j])
  FIM PARA
FIM PARA
